async function init() { const hash = window.location.hash.substring(1); // 去掉 # const params = new URLSearchParams(hash); const accessToken = params.get(\'access_token\'); const type = params.get(\'type\'); if (type !== \'recovery\' || !accessToken) { showError(\'連結無效或已過期，請重新申請重設密碼。\'); return; } // 移除 sb.auth.setSession() 的呼叫。 // Supabase 的 updateUser 函數應該能夠直接使用 URL 中的 access_token 來識別用戶。 // 如果 Supabase 在沒有 setSession 的情況下無法進行 updateUser， // 則需要進一步研究 Supabase 的重設密碼流程，確保其安全性。 // 驗證成功，顯示表單 document.getElementById(\'loading\').classList.add(\'hidden\'); document.getElementById(\'form\').classList.remove(\'hidden\'); }
